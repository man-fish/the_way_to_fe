# 小程序框架构建之Token基类

## Token基类

```js
import { reqUrl } from "../config.js"

class Token{
  constructor() {
    this.getTokenUrl = reqUrl.url+"/token"
    this.verifyTokenUrl = reqUrl.url+"/token/verify"
  }

  verify(){
      let token = wx.getStorageSync('token')
      if(!token){
          this.getToken()
      }else{
        this._verifyToken(token)
      }
  }

  _verifyToken(token) {
    var that = this;
    wx.request({
      url: that.verifyTokenUrl,
      method: 'POST',
      data: {
        token: token
      },
      success: function (res) {
        console.log(res)
        var valid = res.data.isValid;
        if (!valid) {
          that.getToken();
        }
      }
    })
  }

  getToken(callBack) {
    console.log(this.getTokenUrl)
    var that = this;
    wx.login({
      success: function (res) {
        wx.request({
          url: that.getTokenUrl,
          method: 'POST',
          data: {
            account: res.code,
            type: 100
          },
          success: function (res) {
            console.log("我执行了我拿到了token")
            wx.setStorageSync('token', res.data.token)
            callBack && callBack(res.data.token);
          }
        })
      }
    })
  }
}
export {
  Token
}
```

## 调用

```js
//app.js
import {Token} from "utils/token.js"

App({
  onLaunch: function () {
    // 登录
    Token.verify();
  }
})
```
