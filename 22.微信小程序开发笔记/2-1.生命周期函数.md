# 生命周期函数

## app.js

```js
App({

  /**
   * 当小程序初始化完成时，会触发 onLaunch（全局只触发一次），只会在每次刚打开小程序时执行。
   * 从后台切入时不行。
   */
  onLaunch: function () {
      console.log("小程序启动")
      // 获取用户信息,获取主要数据等等
  },

  /**
   * 当小程序启动，或从后台进入前台显示，会触发 onShow
   */
  onShow: function (options) {
      console.log(options) 
      //options中可以获取用户的进入方式
  },

  /**
   * 当小程序从前台进入后台，会触发 onHide
   */
  onHide: function () {
      console.log("用户切入后台")
  },

  /**
   * 当小程序发生脚本错误，或者 api 调用失败时，会触发 onError 并带上错误信息
   */
  onError: function (msg) {

  }
})
```

## index.js

```js
Page({

  /**
   * 页面的初始数据
   */
  data: {

  },

  /**
   * 生命周期函数--监听页面加载
   * 获取具体业务逻辑数据
   * 类似初始化的ajax请求
   */
  onLoad: function (options) {
    	console.log("主页面数据加载")
      	//最重要--获取数据
  },

   /**
   * 生命周期函数--监听页面显示
   */
  onShow: function () {
    	console.log("主页面显示")
  },
    
  /**
   * 生命周期函数--监听页面初次渲染完成
   */
  onReady: function () {
    	//用于渲染完成，弹出提示
  },

  /**
   * 生命周期函数--监听页面隐藏
   */
  onHide: function () {
      console.log('主页面隐藏')
  },

  /**
   * 生命周期函数--监听页面卸载
   */
  onUnload: function () {
      console.log("主页面卸载")
      //tabbar中的页面不会卸载，只有连接跳过去，之后返回退出的页面才有onUnload
  },

  /**
   * 页面相关事件处理函数--监听用户下拉动作
   */
  onPullDownRefresh: function () {
    
  },

  /**
   * 页面上拉触底事件的处理函数
   */
  onReachBottom: function () {
    
  },

  /**
   * 用户点击右上角分享
   */
  onShareAppMessage: function () {
    
  }
})
```

> 因为app.js中没有onready所以大概的执行顺序是：

全局 （onlaunch —— onshow ）—— 具体（ onload —— onshow —— onready ）
